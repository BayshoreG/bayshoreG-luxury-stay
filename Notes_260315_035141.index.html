const express = require("express") const mongoose = require("mongoose") const cors = require("cors") const Booking = require("./bookingModel") const app = express() app.use(cors()) app.use(express.json()) mongoose.connect("mongodb://127.0.0.1:27017/bayshore") app.post("/book", async (req,res)=>{ const booking = new Booking(req.body) await booking.save() res.json({message:"Booking saved"}) }) app.get("/bookings", async(req,res)=>{ const bookings = await Booking.find() res.json(bookings) }) app.listen(5000,()=>{ console.log("Server running on port 5000") })
